1"""Classic function interface to ``pyisbn``. 

2 

3This module supports the calculation of ISBN checksums with 

4``calculate_checksum()``, the conversion between ISBN-10 and ISBN-13 with 

5``convert()`` and the validation of ISBNs with ``validate()``. 

6 

7.. note:: 

8 

9 All the ISBNs must be passed in as ``str`` types, even if it would seem 

10 reasonable to accept some ``int`` forms. The reason behind this is English 

11 speaking countries use ``0`` for their group identifier, and Python 3 would 

12 treat this as a syntax error [#]_. While it may be feasible to allow some 

13 cases as non-``str`` types the complexity in design and usage isn't worth 

14 the minimal benefit. 

15 

16The functions in this module also support 9-digit SBNs for people with older 

17books in their collection. 

18 

19.. [#] Previous Python releases would have assumed it was octal representation 

20 of a number 

21"""

46def calculate_checksum(isbn: TIsbn) -> str: 

47 """Calculate ISBN checksum. 

48 

49 Args: 

50 isbn: SBN, ISBN-10 or ISBN-13 

51 

52 Returns: 

53 Checksum for given ISBN or SBN 

54 

55 """ 

56 digits = [int(i) for i in isbn_cleanse(isbn, checksum=False)] 

57 if len(digits) == _constants.ISBN10_LENGTH_NO_CHECKSUM: 

58 products = [x * y for x, y in enumerate(digits, 1)] 

59 check = sum(products) % _constants.ISBN10_CHECKSUM_MODULUS 

60 if check == _constants.ISBN10_CHECKSUM_X: 

61 check = "X" 

62 else: 

63 products = [ 

64 (x if n % 2 == 0 else x * _constants.ISBN13_ODD_MULTIPLIER) 

65 for n, x in enumerate(digits) 

66 ] 

67 check = ( 

68 _constants.ISBN13_CHECKSUM_SUBTRACT 

69 - sum(products) % _constants.ISBN13_CHECKSUM_MODULUS 

70 ) 

71 if check == _constants.ISBN13_CHECKSUM_MODULUS: 

72 check = _constants.ISBN13_CHECKSUM_TEN_REPLACEMENT 

73 return str(check)

76def convert(isbn: TIsbn, code: str = "978") -> str: 

77 """Convert ISBNs between ISBN-10 and ISBN-13. 

78 

79 Note: 

80 No attempt to hyphenate converted ISBNs is made, because the 

81 specification requires that *any* hyphenation must be correct but 

82 allows ISBNs without hyphenation. 

83 

84 Args: 

85 isbn: SBN, ISBN-10 or ISBN-13 

86 code: EAN Bookland code 

87 

88 Returns: 

89 Converted ISBN-10 or ISBN-13 

90 

91 Raises: 

92 IsbnError: When ISBN-13 isn't convertible to an ISBN-10 

93 

94 """ 

95 isbn = isbn_cleanse(isbn) 

96 if len(isbn) == _constants.ISBN10_LENGTH: 

97 isbn = code + isbn[:-1] 

98 return isbn + calculate_checksum(isbn) 

99 if isbn.startswith(_constants.BOOKLAND_PREFIXES[0]): 

100 return isbn[ 

101 _constants.BOOKLAND_PREFIX_LENGTH : -1 

102 ] + calculate_checksum(isbn[_constants.BOOKLAND_PREFIX_LENGTH : -1]) 

103 raise IsbnError( 

104 "Only ISBN-13s with 978 Bookland code can be converted to ISBN-10." 

105 )

108def validate(isbn: TIsbn) -> bool: 

109 """Validate ISBNs. 

110 

111 Warning: 

112 Publishers have been known to go to press with broken ISBNs, and 

113 therefore validation failures do not completely guarantee an ISBN is 

114 incorrectly entered. It should however be noted that it is massively 

115 more likely *you* have entered an invalid ISBN than the published ISBN 

116 is incorrectly produced. An example of this probability in the real 

117 world is that `Amazon <https://www.amazon.com/>`__ consider it so 

118 unlikely that they refuse to search for invalid published ISBNs. 

119 

120 Args: 

121 isbn: SBN, ISBN-10 or ISBN-13 

122 

123 Returns: 

124 ``True`` if ISBN is valid 

125 

126 """ 

127 isbn = isbn_cleanse(isbn) 

128 return isbn[-1].upper() == calculate_checksum(isbn[:-1])
